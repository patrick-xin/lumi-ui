## Basic Usage

```tsx
import {
  ResizableGroup,
  ResizablePanel,
} from "@components/ui/resizable";

<ResizableGroup orientation="horizontal">
  <ResizablePanel>One</ResizablePanel>
  <ResizablePanel>Two</ResizablePanel>
</ResizableGroup>
```
